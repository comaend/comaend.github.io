import http.server import socketserver import os import cgi PORT = 8080 DIRECTORY = r"C:\Users\Gulbins\Downloads\Neuer Ordner" class MyHandler(http.server.SimpleHTTPRequestHandler): def do_GET(self): # Zeige das Upload-Formular und die Dateiliste self.send_response(200) self.send_header("Content-type", "text/html; charset=utf-8") self.end_headers() files = os.listdir(DIRECTORY) file_list = "" for f in files: if f != "server.py": # Verstecke das Skript selbst file_list += f'
  • {f} [Download]
    • ' html = f""" ZTE Upload Server

    📂 ZTE A56 Datei Server

    Upload Datei:

    Vorhandene Dateien:

      {file_list}
    """ self.wfile.write(html.encode('utf-8')) def do_POST(self): form = cgi.FieldStorage( fp=self.rfile, headers=self.headers, environ={'REQUEST_METHOD': 'POST', 'CONTENT_TYPE': self.headers['Content-Type'], }) item = form['file'] file_path = os.path.join(DIRECTORY, item.filename) with open(file_path, 'wb') as f: f.write(item.file.read()) self.send_response(302) # Weiterleitung nach Upload self.send_header('Location', '/') self.end_headers() os.chdir(DIRECTORY) with socketserver.TCPServer(("", PORT), MyHandler) as httpd: print(f"Server läuft auf Port {PORT}") httpd.serve_forever()
